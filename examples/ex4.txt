PROGRAM Populate Scores Array
// Line numbers are ignored :-)
1. DECLARE Double examScores [] ;
0. DECLARE String nextLine ;
2. DECLARE Integer counter ;
3. SET counter = 0 ;
4. OPEN “examples/exam-scores.xml” ;
5. READ nextLine ;
5. WHILE nextLine != "END"
6. 		SET examScores[counter] = (Double)(nextLine) ;
		DISPLAY examScores[counter] ;
7. 		SET counter = counter + 1 ;
5. 		READ nextLine ;
8. ENDWHILE
9. CLOSE “examples/exam-scores.xml” ;
END
